        // --- Action Functions (Simulation) ---

        /** Simulates downloading the project HTML. */
        function simulateDownload(id) {
            const project = projects.find(p => p.id === id);
            if (project) {
                // In a real app, this would trigger a file download
                alert(`Simulating download of "${project.title}.html"`);
                console.log(`Download content for ${project.title}: \n${project.html}`);
            }
        }

        /** Copies the project HTML code to the clipboard. */
        function copyCode(id) {
            const project = projects.find(p => p.id === id);
            if (project) {
                navigator.clipboard.writeText(project.html).then(() => {
                    alert(`Code for "${project.title}" copied to clipboard!`);
                }).catch(err => {
                    console.error('Could not copy text: ', err);
                    alert('Error: Could not copy code. See console for details.');
                });
            }
        }

        /** Simulates project duplication. */
        function duplicateProject(id) {
            const original = projects.find(p => p.id === id);
            if (original) {
                const newId = projects.reduce((max, p) => Math.max(max, p.id), 0) + 1;
                const duplicated = {
                    ...original,
                    id: newId,
                    title: `${original.title} (Copy)`,
                    createdAt: new Date()
                };
                projects.push(duplicated);
                renderProjects();
                alert(`Project "${original.title}" duplicated as "${duplicated.title}".`);
            }
        }

        /** Simulates adding a new empty project. */
        function simulateAddProject() {
            const newId = projects.reduce((max, p) => Math.max(max, p.id), 0) + 1;
            const newProject = {
                id: newId, 
                title: `New Project ${newId}`, 
                prompt: 'Describe your idea here...', 
                html: '\n\n
Hello World
', 
                createdAt: new Date(), 
                liveUrl: null, 
                hasThumbnail: false
            };
            projects.push(newProject);
            renderProjects();
            openEditModal(newId); // Open the new project directly for editing
        }


        // --- Event Listeners and Initialization ---

        document.addEventListener('DOMContentLoaded', () => {
            
            // Initial render
            renderProjects();

            // 1. Initial Fade-in of Dashboard Content
            setTimeout(() => {
                document.getElementById('loading-overlay').style.opacity = '0';
                document.getElementById('loading-overlay').classList.add('pointer-events-none');
                document.getElementById('dashboard-content').style.opacity = '1';
                document.getElementById('dashboard-content').style.transform = 'translateY(0)';
            }, 800); // Simulate network latency

            // 2. Dark Mode Setup
            const darkModeToggle = document.getElementById('dark-mode-toggle');
            const isDarkMode = localStorage.getItem('darkMode') === 'true';

            if (isDarkMode) {
                document.documentElement.classList.add('dark');
            }

            darkModeToggle.addEventListener('click', () => {
                if (document.documentElement.classList.contains('dark')) {
                    document.documentElement.classList.remove('dark');
                    localStorage.setItem('darkMode', 'false');
                } else {
                    document.documentElement.classList.add('dark');
                    localStorage.setItem('darkMode', 'true');
                }
            });

            // 3. Debounced Search Input
            const searchInput = document.getElementById('search-input');
            const debouncedSearch = debounce((value) => {
                currentSearchTerm = value;
                renderProjects();
            }, 300); // 300ms debounce

            searchInput.addEventListener('input', (e) => {
                debouncedSearch(e.target.value);
            });

            // 4. Sort Dropdown
            document.getElementById('sort-dropdown').addEventListener('change', (e) => {
                currentSort = e.target.value;
                renderProjects();
            });

            // 5. Add Project Button
            document.getElementById('add-project-btn').addEventListener('click', simulateAddProject);
            
            // 6. Prevent click propagation from buttons/modals to parent card (Done inline on buttons)

            // Re-initialize Lucide icons one last time (if any were missed)
            lucide.createIcons();
        });
